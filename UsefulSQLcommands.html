SELECT CONCAT(br.first_name,' ',br.last_name) AS NAME, br.number, 
 CASE WHEN t2.score IS NULL THEN 0 ELSE t2.score END AS score, 
 CASE WHEN t2.topContributor IS NULL THEN 'Not Raced Yet' ELSE t2.topContributor END AS topContributor, 
 CASE WHEN t2.profilePicture IS NULL THEN '' ELSE t2.profilePicture END AS profilePicture 
 FROM bikes_riders AS br
 LEFT JOIN (SELECT 
 t.bikeId, 
     MAX(t.globalScore) AS score, 
     CASE WHEN uf.facebook_first_name IS NULL THEN (CASE WHEN u.first_name <>'' THEN CONCAT(u.first_name,' ',u.last_name) ELSE CONCAT('Guest',u.id) END) ELSE CONCAT(uf.facebook_first_name,' ',uf.facebook_last_name) END AS topContributor, 
     CASE WHEN uf.facebook_id IS NULL THEN '' ELSE CONCAT('https://graph.facebook.com/',uf.facebook_id,'/picture?type=normal') END AS profilePicture 
     FROM (SELECT ub.userId, gs.userBikeId,ub.bikeId,b.name,gs.globalScore
           FROM fwc_global_scores AS gs
           JOIN users_bikes_v16 AS ub ON ub.id=gs.userBikeId
           JOIN bikes AS b ON b.id=ub.bikeId
           AND gs.createdAt > '2021-01-25 00:00:00'
           AND gs.createdAt <= '2021-02-07 23:59:59'
     ORDER BY globalScore DESC 
     LIMIT 20000) AS t
     JOIN users AS u ON u.id=t.userId
     LEFT JOIN users_facebook AS uf ON uf.user_id=u.id
     LEFT OUTER JOIN cheaters AS c ON c.userId=t.userId
     WHERE c.userId IS NULL
     GROUP BY t.bikeId) AS t2 ON t2.bikeId=br.id ORDER BY score DESC;

IAP revenue

/*iap revenue */
SELECT DATE(l.timestamp), SUM(p.cents)/100 FROM iap_logs AS l 
  JOIN iap_prices AS p ON p.product_id=l.product_id 
  WHERE l.success=2 
  GROUP BY DATE(l.timestamp) DESC;
